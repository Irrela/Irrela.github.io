### Description
一个len为M的棋盘，上面有O个障碍，N条行走指令，agent从(0, 0)出发，按指令行走：

1. 每次行走计数器+1，将更新在走到的坐标格上
2. 如果遇到障碍，则坐标不变，但计数器增加
3. 走出棋盘则结束，输出:INVALID

障碍由row 和 col标记位置
指令由 方向 和 step 组成

输入例子：
```
5
3
4
3 0
0 2
2 3
R 2
D 4
R 3
U 3
```
输出例子：
```
0    2    X    0    0
0    3    0    0    12
0    4    0    X    11
X    5    0    0    10
0    6    7    8    9
```
```java
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner in = new Scanner(System.in);
    
        int m = in.nextInt();
        int obstacle_num = in.nextInt();
        int n = in.nextInt();

        long[][] board = new long[m][m];
        
        for(int i = 0; i < obstacle_num; i++){
            int row = in.nextInt();
            int col = in.nextInt();

            board[row][col] = -1;
        }
        
        long count = 0;
        int curRow = 0;
        int curCol = 0;

        for(int i = 0; i < n; i++){
            
            String op = in.next();
            int step = in.nextInt();
            
            for(int j = 0; j < step; j++){
                count++;
                int newRow = curRow;
                int newCol = curCol;

                if(op.equals("R")){
                    newCol++;
                }else if(op.equals("L")){
                    newCol--;
                }else if(op.equals("U")){
                    newRow--;
                }else if(op.equals("D")){
                    newRow++;
                }

                if(newRow < 0 || newRow >= m || newCol < 0 || newCol >= m){
                    System.out.println("INVALID");
                    return;
                }

                if(board[newRow][newCol] == -1){
                    newRow = curRow;
                    newCol = curCol;
                }

                board[newRow][newCol] = count;

                curRow = newRow;
                curCol = newCol;
            }
        }

        for(int i = 0; i < m; i++){
            for(int j = 0; j < m - 1; j++){
                long num = board[i][j];

                if(num < 0){
                    System.out.print("X" + "    ");
                }else if(num < 10){
                    System.out.print(num + "    ");
                }else if(num < 100){
                    System.out.print(num + "   ");
                }else if(num < 1000){
                    System.out.print(num + "  ");
                }else if(num < 10000){
                    System.out.print(num + " ");
                }else if(num < 100000){
                    System.out.print(num);
                }else{
                    System.out.print(99999);
                }

            }

            long endNum = board[i][m - 1];
            if(endNum < 0){
                System.out.print("X\n");
            }else if(endNum >= 100000){
                System.out.print(99999 + "\n");
            }else{
                System.out.print(endNum + "\n");
            }
        }
    }
}
```
